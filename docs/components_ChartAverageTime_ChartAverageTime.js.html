import React, { PureComponent } from 'react'
import { LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, Legend, ResponsiveContainer } from 'recharts'
import './ChartAverageTime.css'
import CustomToolTip from './CustomToolTip'
import CustomCursor from './CustomCursor'
import PropTypes from 'prop-types'

/**
 * Returns a line-chart component from fetched data, giving an overview over the week trainings
 * @param {array} donnees The fetched data including 'day' and 'sessionLength'
 * @param {number} day -
 * @param {number} sessionLength -
 * @returns {React.ReactElement} -
 */


function ChartAverageTime({donnees}) {
  
  return (
    <div id='averageTimeFullChart' style={{height:'100%',width:'100%'}}>
      <p id='averageTimeTitle'>Durée moyenne des sessions</p>
      <div id='averageTimeDays'>
        <p>L</p>
        <p>M</p>
        <p>M</p>
        <p>J</p>
        <p>V</p>
        <p>S</p>
        <p>D</p>
      </div>
      <ResponsiveContainer width="100%" height="100%">
        <LineChart
          width={500}
          height={250}
          data={donnees}
          margin={{
            bottom: 10
          }}
        >
          <CartesianGrid strokeDasharray="3 3" horizontal={false} vertical={false}/>
          <XAxis hide dataKey="day" tickLine={false} axisLine={false} />
          <YAxis hide={true} padding={{top:80, bottom: 100}} />
          <Tooltip wrapperStyle={{ display:'flex', backgroundColor:"white", color: "black"}}
            labelStyle={{display: "none"}}
            content={<CustomToolTip/>}
            cursor={<CustomCursor />} />
          <Line type="natural" dataKey="sessionLength" activeDot={{ stroke: "rgba(255, 255, 255, 0.5)", strokeWidth: 10, r: 5, fill:'#fff'}} 
          stroke="#fff" 
          strokeWidth={3} dot={false}/>

        </LineChart>
      </ResponsiveContainer>
    </div>
  );
  
}

ChartAverageTime.propTypes = {
  donnees: PropTypes.arrayOf(
    PropTypes.shape({
      day: PropTypes.number.isRequired,
      sessionLength: PropTypes.number.isRequired,
    }).isRequired,
  )
}

export default ChartAverageTime
